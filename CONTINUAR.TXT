<script type="text/javascript">
$(document).ready(function() {
    if ( $("select[name=provincia]").val() === '' ) {
        $("select[name=departamento]").prop('disabled', 'false');
        $("select[name=localidad]").prop('disabled', 'false');
    }
});

$("select[name=provincia]").change(function() {
    $("select[name=departamento]").prop('disabled', $(this).val() === '');
});

$("select[name=departamento]").change(function() {
    $("select[name=localidad]").prop('disabled', $(this).val() === '');
    $($(this).val()).prop('disabled', $("select[name=provincia]").val() === '');
});

$("select[name=localidad]").change(function() {
    $($(this).val()).prop('disabled', $("select[name=departamento]").val() === '');
});

</script>
